
func TestAccCableResource_externalDeletion(t *testing.T) {
	t.Parallel()

	testutil.TestAccPreCheck(t)

	// Simple cable with minimal setup
	resource.Test(t, resource.TestCase{
		PreCheck:                 func() { testutil.TestAccPreCheck(t) },
		ProtoV6ProviderFactories: testutil.TestAccProtoV6ProviderFactories,

		Steps: []resource.TestStep{
			{
				Config: `
resource "netbox_site" "test" {
  name = "test-cable-site"
  slug = "test-cable-site"
}

resource "netbox_manufacturer" "test" {
  name = "test-cable-mfg"
  slug = "test-cable-mfg"
}

resource "netbox_device_type" "test" {
  manufacturer = netbox_manufacturer.test.name
  model = "test-cable-model"
  slug = "test-cable-model"
}

resource "netbox_device_role" "test" {
  name = "test-cable-role"
  slug = "test-cable-role"
}

resource "netbox_device" "test_a" {
  name = "test-cable-device-a"
  device_type = netbox_device_type.test.model
  role = netbox_device_role.test.name
  site = netbox_site.test.name
}

resource "netbox_device" "test_b" {
  name = "test-cable-device-b"
  device_type = netbox_device_type.test.model
  role = netbox_device_role.test.name
  site = netbox_site.test.name
}

resource "netbox_interface" "test_a" {
  name = "eth0"
  device = netbox_device.test_a.name
  type = "1000base-t"
}

resource "netbox_interface" "test_b" {
  name = "eth0"
  device = netbox_device.test_b.name
  type = "1000base-t"
}

resource "netbox_cable" "test" {
  label = "test-cable-ext-del"
  type = "cat6"

  a_terminations {
    object_type = "dcim.interface"
    object_id = netbox_interface.test_a.id
  }

  b_terminations {
    object_type = "dcim.interface"
    object_id = netbox_interface.test_b.id
  }
}
`,
				Check: resource.ComposeAggregateTestCheckFunc(
					resource.TestCheckResourceAttr("netbox_cable.test", "label", "test-cable-ext-del"),
					resource.TestCheckResourceAttrSet("netbox_cable.test", "id"),
				),
			},
			{
				PreConfig: func() {
					// Simulate external deletion by directly calling the API
					client, err := testutil.GetSharedClient()
					if err != nil {
						t.Fatalf("Failed to get shared client: %v", err)
					}

					// Get the cable by label and delete it
					cables, _, err := client.DcimAPI.DcimCablesList(context.Background()).LabelIc("test-cable-ext-del").Execute()
					if err != nil || cables == nil || len(cables.Results) == 0 {
						t.Fatalf("Failed to find cable for external deletion: %v", err)
					}

					// Delete the first matching cable
					cableID := cables.Results[0].Id
					_, err = client.DcimAPI.DcimCablesDestroy(context.Background(), cableID).Execute()
					if err != nil {
						t.Fatalf("Failed to externally delete cable: %v", err)
					}

					t.Logf("Successfully externally deleted cable with ID: %d", cableID)
				},
				Config: `
resource "netbox_site" "test" {
  name = "test-cable-site"
  slug = "test-cable-site"
}

resource "netbox_manufacturer" "test" {
  name = "test-cable-mfg"
  slug = "test-cable-mfg"
}

resource "netbox_device_type" "test" {
  manufacturer = netbox_manufacturer.test.name
  model = "test-cable-model"
  slug = "test-cable-model"
}

resource "netbox_device_role" "test" {
  name = "test-cable-role"
  slug = "test-cable-role"
}

resource "netbox_device" "test_a" {
  name = "test-cable-device-a"
  device_type = netbox_device_type.test.model
  role = netbox_device_role.test.name
  site = netbox_site.test.name
}

resource "netbox_device" "test_b" {
  name = "test-cable-device-b"
  device_type = netbox_device_type.test.model
  role = netbox_device_role.test.name
  site = netbox_site.test.name
}

resource "netbox_interface" "test_a" {
  name = "eth0"
  device = netbox_device.test_a.name
  type = "1000base-t"
}

resource "netbox_interface" "test_b" {
  name = "eth0"
  device = netbox_device.test_b.name
  type = "1000base-t"
}

resource "netbox_cable" "test" {
  label = "test-cable-ext-del"
  type = "cat6"

  a_terminations {
    object_type = "dcim.interface"
    object_id = netbox_interface.test_a.id
  }

  b_terminations {
    object_type = "dcim.interface"
    object_id = netbox_interface.test_b.id
  }
}
`,
				// After applying the config again, Terraform will refresh and detect the 404,
				// triggering the removal from state. The resource will be recreated.
				Check: resource.ComposeAggregateTestCheckFunc(
					resource.TestCheckResourceAttrSet("netbox_cable.test", "id"),
				),
			},
		},
	})
}
